## <a name="example"></a><span data-ttu-id="32e59-109">示例</span><span class="sxs-lookup"><span data-stu-id="32e59-109">Example</span></span>

    <Extension()> _  
    Module StringExtensions  
        <Extension()> _

        End Sub  
    End Module  
End Namespace
