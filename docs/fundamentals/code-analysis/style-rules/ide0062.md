### <a name="example"></a><span data-ttu-id="9cab4-127">示例</span><span class="sxs-lookup"><span data-stu-id="9cab4-127">Example</span></span>

    }

{
    Hello();

    {
        Console.WriteLine("Hello");
    }
}
